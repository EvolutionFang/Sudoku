   5 |  * A node in the backtracking history tree.
   6 |  * Each node represents a snapshot of the game state at a specific moment.

  11 |    * @param {TreeNode|null} parent - Parent node in the history tree.

  14 |    * @param {Object} infos.cell - Cell position {x, y}.

  17 |    */

  20 |     this.historyNode = historyNode;
  21 |     this.parent = parent;
  22 |     this.children = [];

  25 |     this.infos = infos;
  26 |   }

  30 |  * Manager class that maintains a tree-structured history
  31 |  * for backtracking and branching gameplay states.

  35 |     this.root = null;
  36 |     this.currentTreeNode = null;
  37 |     this.isInBacktrackMode = false;
  38 |     this.nodeCount = 0;
  39 |   }

  41 |   /**

  53 |     return this.root;

  57 |    * Save the current game state as a new node in the history tree.
  58 |    * This always creates a child node of the current node.
  59 |    *
  60 |    * @param {string} description - Description of the user action.
  61 |    * @param {boolean} backtrackable - Whether this node is a backtracking decision point.
  62 |    * @param {Object|null} infos - Backtracking metadata (cell, candidates, tried).
  63 |    * @returns {TreeNode|null} The newly created tree node.
  64 |    */

  74 |       infos
